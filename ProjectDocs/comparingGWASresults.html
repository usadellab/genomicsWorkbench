import pandas as pd

def sigHits(df, colName, threshold):
    return sum(df[colName] < threshold)

association_thresholds = [5e-8, 5e-7, 5e-6, 5e-5]

results = []

for threshold in association_thresholds:
    results.append({
        "Threshold": threshold,
        "Fastlmm": sigHits(fastlmmData, "PValue", threshold),
        "Plink1.07_CLI_without": sigHits(plink107DataWithout, "P", threshold),
        "Plink1.07_CLI_with": sigHits(plink107DataWith, "P", threshold),
        "Plink1.9_CLI_without": sigHits(plink19DataWithout, "P", threshold),
        "Plink1.9_CLI_with": sigHits(plink19DataWith, "P", threshold),
        "Plink1.07_WEB_without": sigHits(plink107_Web_Without, "P", threshold),
        "Plink1.07_WEB_with": sigHits(plink107_Web_With, "P", threshold)
        
    })

results_df = pd.DataFrame(results)

Table 1: The above table compares the number of SNPs that reached significane at a given association threshold (AT). Fastlmm results are corrected for population structure using relationship matrix, which is inernally calculated by fastlmm package. In the headers suffixes with and without stand for GWAS runs with and without correction for population structure. The results indicate that default value of AT (5*10^-8) yielded same results when corrected for population structure for all methods. An AT of 5*10^-07 also produced similar results for population corrected GWAS, however fastlmm was more stringent. At more relaxed values of AT yield varying results at respective implementations.

import pandas as pd
from matplotlib import pyplot as plt
from matplotlib import pyplot
from upsetplot import plot
from upsetplot import from_contents


def sigSNPlist(df, colName, threshold):
    df = df[df[colName] < threshold]
    return list(df.SNP)

association_thresholds = [5e-8, 5e-7, 5e-6, 5e-5]

data = dict()

for threshold in association_thresholds:
        data[threshold] = {}
       
        data[threshold]["Fastlmm"] = sigSNPlist(fastlmmData, "PValue", threshold)
        data[threshold]["Plink1.07_CLI_without"] = sigSNPlist(plink107DataWithout, "P", threshold)
        data[threshold]["Plink1.07_CLI_with"] = sigSNPlist(plink107DataWith, "P", threshold)
        data[threshold]["Plink1.9_CLI_without"] = sigSNPlist(plink19DataWithout, "P", threshold)
        data[threshold]["Plink1.9_CLI_with"] = sigSNPlist(plink19DataWith, "P", threshold)
        data[threshold]["Plink1.07_WEB_without"] = sigSNPlist(plink107_Web_Without, "P", threshold)
        data[threshold]["Plink1.07_WEB_with"] = sigSNPlist(plink107_Web_With, "P", threshold)
def createUpsetPlot(threshold):   
    # Your data
    sets = data[threshold]
    dataUpsetPlot = from_contents({'Fastlmm': sets['Fastlmm'],
                          'Plink1.07_CLI_without': sets['Plink1.07_CLI_without'],
                          'Plink1.07_CLI_with': sets['Plink1.07_CLI_with'],
                          'Plink1.9_CLI_without': sets['Plink1.9_CLI_without'],
                          'Plink1.9_CLI_with': sets['Plink1.9_CLI_with'],
                          'Plink1.07_WEB_without': sets['Plink1.07_WEB_without'],
                          'Plink1.07_WEB_with': sets['Plink1.07_WEB_with'],
                         })


    plot(dataUpsetPlot)
    plt.title(f'Intersection of results at pvalues threshold of {threshold}')
    pyplot.show()
